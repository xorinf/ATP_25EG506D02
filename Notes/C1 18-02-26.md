## 1. Conditional Logic
> finding the biggest number using if-else logic

```javascript
// finding big number in 2 numbers
let a = 10;
let b = 20;

if (a > b) {
  console.log(`a is bigger`);
} else {
  console.log(`b is bigger`);
}
```

```javascript
// finding big number in 3 numbers using logical operators
let c = 30;

if (a > b && a > c) {
  console.log(`a is biggest`);
} 
else if (b > a && b > c) {
  console.log(`b is biggest`);
} 
else {
  console.log(`c is biggest`);
}
```

## 2. Arrays and Loops
> working with lists of data

```javascript
// finding sum of marks
let marks = [98, 75, 65, 90, 19];
let sum = 0;

for (let index = 0; index < marks.length; index++){
    sum = sum + marks[index];
}
console.log("total sum is", sum);
```

```javascript
// finding smallest number (min)
let min = marks[0];

for (let index = 0; index < marks.length; index++){
    if (marks[index] < min){
        min = marks[index];
    }
}
console.log("minimum value is", min);
```

## 3. Functions
> wrapping logic into reusable blocks

```javascript
// function to find max of 3 numbers
function threeMax(a, b, c) {
  if (a > b && a > c) return a;
  else if (b > a && b > c) return b;
  else return c;
}

console.log("Big element is :", threeMax(10, 20, 30));
```

```javascript
// function to sum an array
function sumArr(an_array){
    let sum = 0;
    for (let index = 0; index < an_array.length; index++){
        sum = sum + an_array[index];
    }
    return sum;
}
```

```javascript
// linear search function
// checks if key exists in array and returns index
function searchArr(an_array, key) {
  for (let index = 0; index < an_array.length; index++) {
    if (an_array[index] == key) {
      return index; // found
    }
  }
  return -1; // not found
}
```

> note: functions help us reuse code without rewriting it @xorinf
