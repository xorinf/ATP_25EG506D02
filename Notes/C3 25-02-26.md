let arr = [90, 45, -12, 65, 73];

//Filter()
// filtlering means selection! it dosent do any modification
let filtered_r1 = arr.filter((element) => element > 30);
console.log(filtered_r1);

let filtered_r = arr.filter((element) => element >= 40 && element <= 80);
console.log(filtered_r);

//Map()
// this can be used for modification dont use filter there?<
let r3 = arr.map((element) => element + 10);
console.log(r3);

// add +10 elements <50 and -20 from >50

let result = arr.map((element) => {
  if (element < 50) {
    return element + 10;
  } else {
    return element - 20;
  }
});

//Reduce()
const ya_result = arr.reduce((acc, element) => acc + element)
console.log(ya_result)

// finding smallest element using reduce
const small_elements = arr.reduce((acc, element) => {
    if (acc < element){
        return acc
    }else{
        return element
    }
})

// finding largest number using reduce
const large_element = arr.reduce((acc, element) => {
    if (acc > element){
        return acc
    }else{
        return element
    }
})
