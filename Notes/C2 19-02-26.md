```javascript
// main funtion
let makePayment = function(amount, paymentType, platform){
    console.log(`Payment of Rs: ${amount}/- initiated`);
    paymentType(amount, platform)

    console.log(`UPI Payment of RS: ${amount}/- is DONE! using ${platform}\n`);

let sum = function(x){
    // x is always there until the very end 
    return function(y){
        return x + y
    }
}
//closure : invoked -> all the varialble should be variables (only works when func returns func)
// here after this line x is stored in memory
// every js funtion has closure by default because of this closure property the variables of outside scope
// will be maintained temporarily in the heap even the outside funtion execution is completed
let x = sum(20)
// the x will be reserved until whole script executes
console.log(x(10));
```

```javascript
let student = {
  roll: 1,
  name: "Ravi",
  lname: "Kumar",
  marks: [10, 20, 30, 40, 60, 67],
  newfun : function(){
        let temp = this.marks

        let sum = 0
        for (let x in temp){
            sum = sum + temp[x]
        }
        let avg = sum / count
        return avg
  },

    street: 5,
    colony: "venkatapur",

      console.log("YOI");

        let temp = student.marks
        let count = 0
        let sum = 0
        for (let x in temp){
            sum = sum + temp[x]
            count = count + 1
        }

        let avg = sum / count
        return avg
        
    },

      return student.name + " " + student.lname;
    },
  },
```
